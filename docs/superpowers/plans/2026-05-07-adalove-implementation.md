# adalove Implementation Plan

> **For agentic workers:** Steps use checkbox (`- [ ]`) syntax for tracking. Complete every step before moving to the next task.

**Goal:** Build a local Python CLI tool that fetches activities from the Adalove university API, lets the user filter interactively by week and subject via keyboard/mouse menus, and writes two structured markdown files — one with full activity detail and one with links only.

**Architecture:** Strict layered package: `adalove/` contains independent modules (`api`, `models`, `filters`, `output`, `config`) with no circular imports. A `cli/` package owns the entry point and orchestrates everything. A `pyproject.toml` makes `adalove` runnable as a system command after `pip install -e .`.

**Tech Stack:** Python 3.11+, `typer` (CLI), `questionary` (interactive menus), `requests` (HTTP), `html.parser` stdlib (HTML stripping), `pytest` + `pytest-mock` (tests), `setuptools` (packaging).

import re
from dataclasses import dataclass
from html.parser import HTMLParser


class _HTMLStripper(HTMLParser):
    def __init__(self):
        super().__init__()
        self._parts: list[str] = []

    def handle_data(self, data: str) -> None:
        self._parts.append(data)

    def get_text(self) -> str:
        # join without separator then collapse whitespace — avoids double spaces
        # from adjacent inline elements that already carry trailing spaces
        return re.sub(r"\s+", " ", "".join(self._parts)).strip()


def strip_html(html: str | None) -> str:
    if html is None or not isinstance(html, str):
        return ""
    stripper = _HTMLStripper()
    stripper.feed(html)
    return stripper.get_text()


def _parse_folder_number(folder_caption: str) -> int:
    for part in reversed(folder_caption.strip().split()):
        try:
            return int(part)
        except ValueError:
            continue
    return 0  # 0 means unparseable; callers must skip folder_number == 0


@dataclass
class Activity:
    uuid: str
    caption: str
    description: str
    url: str
    professor_name: str
    folder_caption: str
    folder_number: int  # 0 = unparseable caption, excluded from week menus
    study_type: str
    status: int

    @classmethod
    def from_api(cls, data: dict) -> "Activity":
        folder_caption = data.get("folderCaption") or ""
        return cls(
            uuid=data.get("studentActivityUuid") or "",
            caption=data.get("caption") or "",
            description=strip_html(data.get("description")),
            url=data.get("basicActivityURL") or "",
            professor_name=data.get("professorName") or "",
            folder_caption=folder_caption,
            folder_number=_parse_folder_number(folder_caption),
            study_type=data.get("study_type") or "",
            status=data.get("status") or 0,
        )
```

import json
from pathlib import Path

CONFIG_PATH = Path.cwd() / "config.json"


def load_config() -> dict:
    if not CONFIG_PATH.exists():
        raise FileNotFoundError(
            "config.json not found. Run 'adalove setup' first."
        )
    with CONFIG_PATH.open(encoding="utf-8") as f:
        try:
            return json.load(f)
        except json.JSONDecodeError as e:
            raise ValueError(f"config.json is malformed: {e}") from e


def save_config(data: dict) -> None:
    with CONFIG_PATH.open("w", encoding="utf-8") as f:
        json.dump(data, f, indent=2, ensure_ascii=False)

import requests
from adalove.models.activity import Activity

SESSION_EXPIRED_MESSAGE = (
    "Session expired. Run 'adalove setup' to update your token."
)


class AdaloveClient:
    def __init__(self, api_url: str, token: str) -> None:
        self._api_url = api_url
        self._session = requests.Session()
        self._session.headers.update({"Authorization": token})

    def fetch_activities(self) -> list[Activity]:
        try:
            response = self._session.get(self._api_url, timeout=30)
        except requests.RequestException as e:
            raise ConnectionError(str(e)) from e

        if response.status_code in (401, 403):
            raise PermissionError(SESSION_EXPIRED_MESSAGE)

        response.raise_for_status()

        data = response.json()
        return [Activity.from_api(a) for a in data.get("activities", [])]
```

```python
from adalove.models.activity import Activity

_NAO_PRESENTE = "Não presente no módulo"


def filter_activities(
    activities: list[Activity],
    weeks: list[int],
    subjects: list[str],
    teacher_subjects: dict[str, str],
) -> list[Activity]:
    result = []
    for activity in activities:
        if activity.folder_number == 0:  # unparseable caption, skip
            continue
        if weeks and activity.folder_number not in weeks:
            continue
        mapped = teacher_subjects.get(activity.professor_name, _NAO_PRESENTE)
        if mapped == _NAO_PRESENTE:
            continue
        if subjects and mapped not in subjects:
            continue
        result.append(activity)
    return result


def get_unique_weeks(activities: list[Activity]) -> list[tuple[int, str]]:
    seen: dict[int, str] = {}
    for a in activities:
        if a.folder_number == 0:  # skip unparseable captions
            continue
        if a.folder_number not in seen:
            seen[a.folder_number] = a.folder_caption
    return sorted(seen.items())


def get_unique_teachers(activities: list[Activity]) -> list[str]:
    return sorted({a.professor_name for a in activities if a.professor_name})
```

from collections import defaultdict
from datetime import date
from pathlib import Path

from adalove.models.activity import Activity

OUTPUT_DIR = Path.cwd() / "output"


def write_links_md(
    activities: list[Activity],
    teacher_subjects: dict[str, str],
    selected_weeks: list[int],
    selected_subjects: list[str],
) -> Path:
    OUTPUT_DIR.mkdir(exist_ok=True)
    path = OUTPUT_DIR / "links.md"

    weeks_label = ", ".join(f"Semana {w:02d}" for w in sorted(selected_weeks)) or "Todas"
    subjects_label = ", ".join(sorted(selected_subjects)) or "Todas"
    today = date.today().isoformat()

    grouped: dict[int, dict[str, list[Activity]]] = defaultdict(lambda: defaultdict(list))
    for a in activities:
        if not a.url:
            continue
        subject = teacher_subjects.get(a.professor_name, "")
        grouped[a.folder_number][subject].append(a)

    lines: list[str] = [
        f"# Links — {weeks_label} | {subjects_label}",
        f"> Gerado em: {today}",
        "",
    ]

    for week_num in sorted(grouped):
        week_caption = next(
            (a.folder_caption for a in activities if a.folder_number == week_num),
            f"Semana {week_num:02d}",
        )
        lines.append(f"## {week_caption}")
        lines.append("")

        for subject in sorted(grouped[week_num]):
            lines.append(f"### {subject}")
            for activity in grouped[week_num][subject]:
                lines.append(f"- [{activity.caption}]({activity.url})")
            lines.append("")

    path.write_text("\n".join(lines), encoding="utf-8")
    return path

```python
@app.command()
def fetch() -> None:
    """Interactively filter activities and write output markdown files."""
    try:
        config = load_config()
    except FileNotFoundError as e:
        typer.echo(str(e), err=True)
        raise typer.Exit(1)

    teacher_subjects: dict[str, str] = config.get("teacher_subjects", {})

    typer.echo("Fetching activities...")
    try:
        client = AdaloveClient(api_url=config["api_url"], token=config["token"])
        activities = client.fetch_activities()
    except KeyError as e:
        typer.echo(f"config.json is missing key {e}. Run 'adalove setup' again.", err=True)
        raise typer.Exit(1)
    except PermissionError as e:
        typer.echo(str(e), err=True)
        raise typer.Exit(1)
    except ConnectionError as e:
        typer.echo(f"Connection error: {e}", err=True)
        raise typer.Exit(1)

    typer.echo(f"✓ {len(activities)} activities loaded.\n")

    week_choices = [
        questionary.Choice(title=caption, value=num)
        for num, caption in get_unique_weeks(activities)
    ]

    # Re-prompt until the user picks at least one week, or cancels with Ctrl+C
    while True:
        selected_week_nums = questionary.checkbox(
            "Select weeks (Space to toggle, Enter to confirm):",
            choices=week_choices,
        ).ask()
        if selected_week_nums is None:  # Ctrl+C
            typer.echo("Cancelled.")
            raise typer.Exit(0)
        if selected_week_nums:
            break
        typer.echo("Select at least one week.")

    subject_choices = [s for s in SUBJECTS if s != "Não presente no módulo"]

    # Re-prompt until the user picks at least one subject, or cancels with Ctrl+C
    while True:
        selected_subjects = questionary.checkbox(
            "Select subjects:",
            choices=subject_choices,
        ).ask()
        if selected_subjects is None:  # Ctrl+C
            typer.echo("Cancelled.")
            raise typer.Exit(0)
        if selected_subjects:
            break
        typer.echo("Select at least one subject.")

    filtered = filter_activities(
        activities=activities,
        weeks=selected_week_nums,
        subjects=selected_subjects,
        teacher_subjects=teacher_subjects,
    )

    if not filtered:
        typer.echo("\nNo activities match the selected filters.")
        raise typer.Exit(0)

    typer.echo(f"\n{len(filtered)} activities matched. Writing files...")

    activities_path = write_activities_md(
        activities=filtered,
        teacher_subjects=teacher_subjects,
        selected_weeks=selected_week_nums,
        selected_subjects=selected_subjects,
    )
    links_path = write_links_md(
        activities=filtered,
        teacher_subjects=teacher_subjects,
        selected_weeks=selected_week_nums,
        selected_subjects=selected_subjects,
    )

    typer.echo(f"✓ Written: {activities_path}")
    typer.echo(f"✓ Written: {links_path}")
    typer.echo("Note: previous output/activities.md and output/links.md were overwritten.")
```
